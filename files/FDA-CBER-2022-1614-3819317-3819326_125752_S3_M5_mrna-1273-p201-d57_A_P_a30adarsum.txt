proc sort data=adb.adsl out=adsl; by studyid usubjid; run;

data ar1;
	merge ADSL (in=ina) faae (in=inface);
	by studyid usubjid;	
	if ina and inface;
	if faobj in ('Erythema','Swelling') and index(upcase(fatest), 'OCCURRENCE') > 0 then delete;
	if faobj = 'Receive Medical Attention?' and index(upcase(fatest), 'OCCURRENCE') > 0 then delete;
	length ATPTREF $50;
	if index(upcase(fatptref),'DOSE 1') > 0 then ATPTREF = 'Vaccination 1';
	else if index(upcase(fatptref),'DOSE 2') > 0 then ATPTREF = 'Vaccination 2';
	if facat = 'REACTOGENICITY';
	
	
	/*for AVAL derivation only, will not be kept in final dataset*/	
	if index(upcase(FASTRESC),"GRADE 0") > 0 then ATOXGR = 'Grade 0';
	if index(upcase(FASTRESC),"GRADE 1") > 0 then ATOXGR = 'Grade 1';
	if index(upcase(FASTRESC),"GRADE 2") > 0 then ATOXGR = 'Grade 2'; 
	if index(upcase(FASTRESC),"GRADE 3") > 0 then ATOXGR = 'Grade 3';
	if index(upcase(FASTRESC),"GRADE 4") > 0 then ATOXGR = 'Grade 4';
run;

data ar2;
	merge adsl (in=ina) vs (in=invs drop=subjid);
	by studyid usubjid;
	if ina and invs;
	if vscat = 'REACTOGENICITY' and  vstptnum > 7;
	length ATPTREF $50;
	if index(upcase(vstptref),'DOSE 1') > 0 then ATPTREF = 'Vaccination 1';
	else if index(upcase(vstptref),'DOSE 2') > 0 then ATPTREF = 'Vaccination 2';
run;

%macro create_paramcd1(paramcd=, param=, subset=, subcond=);
data adar_&paramcd;
	set adar;
	where paramcd = "&subset." &subcond.;
run;
proc sort nodupkey; by usubjid atptref atpt  ; run;

proc sql noprint; 
	create table _&paramcd. as select distinct studyid, usubjid, atptref,
		"&paramcd." as paramcd, "&param." as param, count (*) as aval
		from adar_&paramcd  
		group by usubjid, atptref;
quit;

%mend create_paramcd1;

%macro create_paramcd2(paramcd=, param=, subset=, subcond=%str(and atoxgrn>0), aedata=ar1, varlist1=faobj fatest fastresc fastresn, varlist2=fatpt, aecond=);
proc sort data=adar out=adar_&paramcd;
	by usubjid atptref atptgr1 atoxgrn atptn;
	where paramcd = "&subset." &subcond.; 
run;

data adar_&paramcd.;
	set adar_&paramcd.;
	by usubjid atptref atptgr1 atoxgrn atptn ;
	if last.atptgr1;
run;

proc sql noprint; 
	create table adar_&paramcd._cnt as select distinct studyid, usubjid, atptref, count (*) as totcnt
		from adar_&paramcd. 
		where paramcd = "&subset." &subcond.
		group by usubjid, atptref;
quit;

Proc sort data=&aedata. out=&aedata._&paramcd. (keep=Studyid usubjid atptref &varlist1. &varlist2.) nodupkey;
	by studyid usubjid atptref &varlist2.;
	where &aecond.;
run;

proc sql noprint; 
	create table &aedata._&paramcd._cnt as select distinct studyid, usubjid, atptref, count (*) as totcnt2
		from &aedata._&paramcd. 
		group by usubjid, atptref;
quit;

data adar_&paramcd._cnt_&aedata.;
	merge adar_&paramcd._cnt (in=a) &aedata._&paramcd._cnt (in=b);
	by Studyid usubjid atptref;
	if a or b;

        if totcnt ne . and totcnt2 ne . then aval = sum(totcnt, totcnt2);
	else if totcnt ne . and totcnt2 eq . then aval = totcnt;
	else if totcnt eq . and totcnt2 ne . then do;
		%if &aedata.= ar1 %then %do;
			put 'ALERT_C: subject there is duration in FAAE but not in ADAR ' USUBJID  ' ' ATPTREF;
			%end;
		%else %if &aedata= ar2 %then %do;
			put 'ALERT_C: subject there is duration in VS but not in ADAR ' USUBJID  ' ' ATPTREF;
			%end;
		aval = totcnt2;
	     end;
run;

proc sql noprint; 
	create table _&paramcd. as select distinct studyid, usubjid, atptref,
		"&paramcd." as paramcd, "&param." as param, aval
		from adar_&paramcd._cnt_&aedata. 
		order by usubjid, atptref;
quit;

%mend create_paramcd2;

%macro create_paramcd3(paramcd=, param=, subcond1=, subcond2=, aecond1=, aecond2=);
proc sort data=adar out=adar_&paramcd;
	by usubjid atptref atptgr1 atoxgrn atptn;
	where &subcond1. &subcond2.;
run;

data adar_&paramcd.;
	set adar_&paramcd.;
	by usubjid atptref atptgr1 atoxgrn atptn ;
	if last.atptgr1;
run;
 
proc sql noprint; 
	create table adar_&paramcd._cnt as select distinct studyid, usubjid, atptref, count (*) as totcnt
		from adar_&paramcd. 
		where &subcond1. &subcond2.
		group by usubjid, atptref;
quit;

Proc sort data=ar1 out=ar1_&paramcd. (keep=Studyid usubjid atptref faobj fatest fastresc fastresn fatpt fascat) nodupkey;
	by studyid usubjid atptref fatpt;
	where &aecond1.;
run;
 
proc sql noprint; 
	create table ar1_&paramcd._cnt as select distinct studyid, usubjid, atptref, count (*) as totcnt2
		from ar1_&paramcd. 
		group by usubjid, atptref;
quit;

Proc sort data=ar2 out=ar2_&paramcd. (keep=Studyid usubjid atptref vstestcd vsstresn vstpt) nodupkey;
	by studyid usubjid atptref vstpt;
	where &aecond2.;
run;
 

proc sql noprint; 
	create table ar2_&paramcd._cnt as select distinct studyid, usubjid, atptref, count (*) as totcnt3
		from ar2_&paramcd. 
		group by usubjid, atptref;
quit;

data adar_&paramcd._cnt_ar;
	merge adar_&paramcd._cnt (in=a) ar1_&paramcd._cnt (in=b) ar2_&paramcd._cnt (in=c);
	by Studyid usubjid atptref;
	if a or b or c;
        if totcnt ne . or totcnt2 ne . or totcnt3 ne . then aval = sum(totcnt, totcnt2, totcnt3);
run;

proc sql noprint; 
	create table _&paramcd. as select distinct studyid, usubjid, atptref,
		"&paramcd." as paramcd, "&param." as param, aval
		from adar_&paramcd._cnt_ar 
		order by usubjid, atptref;
quit;

%mend create_paramcd3;

%macro create_paramcd4(paramcd=, param=, subset=, subcond=%str(and not missing(avalc)));
data _&paramcd.;	
	set adar;
	where paramcd = "&subset."  &subcond.;
run;

proc sort data=_&paramcd.; by usubjid atptref avalc descending aseq; run;

data _&paramcd.;
	set _&paramcd.;
	by usubjid atptref avalc descending aseq;
	if last.atptref;
	keep studyid usubjid atptref avalc aseq;
run;

data _&paramcd.;
	set _&paramcd.;
	paramcd = "&paramcd.";
	param = "&param.";
run;
	
%mend create_paramcd4;
